# Write your own code
def totalstats(TotalStats):
    if TotalStats > 85:
        classification = "Top Performer"
    elif TotalStats <= 85:
        classification = "Non Top-Peformer"
    return classification

players["player_classification"] = players["TotalStats"].apply(totalstats)
